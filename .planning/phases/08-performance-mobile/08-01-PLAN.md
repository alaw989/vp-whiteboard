       function getViewportBounds(containerWidth: number, containerHeight: number) {
         const { x, y, zoom } = viewport.value
         return {
           left: -x / zoom,
           top: -y / zoom,
           right: (-x + containerWidth) / zoom,
           bottom: (-y + containerHeight) / zoom,
         }
       }
