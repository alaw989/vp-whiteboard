  <name>Create useKeyboardShortcuts composable</name>
  <files>composables/useKeyboardShortcuts.ts</files>
  <action>
    Create `composables/useKeyboardShortcuts.ts` with global keyboard handler:

    ```typescript
    import { onMounted, onUnmounted } from 'vue'

    export interface KeyboardShortcutOptions {
      canUndo: () => boolean
      canRedo: () => boolean
      onUndo: () => void
      onRedo: () => void
      onDelete?: () => void
    }

    export function useKeyboardShortcuts(options: KeyboardShortcutOptions) {
      const { canUndo, canRedo, onUndo, onRedo, onDelete } = options

      function handleKeydown(event: KeyboardEvent) {
        // Check if user is typing in an input/textarea - don't trigger shortcuts
        const target = event.target as HTMLElement
        if (target.tagName === 'INPUT' ||
            target.tagName === 'TEXTAREA' ||
            target.contentEditable === 'true') {
          return
        }

        // Ctrl+Z or Cmd+Z for undo
        if ((event.ctrlKey || event.metaKey) && event.key === 'z' && !event.shiftKey) {
          event.preventDefault()
          if (canUndo()) {
            onUndo()
          }
          return
        }

        // Ctrl+Y or Cmd+Y for redo
        if ((event.ctrlKey || event.metaKey) && event.key === 'y') {
          event.preventDefault()
          if (canRedo()) {
            onRedo()
          }
          return
        }

        // Ctrl+Shift+Z or Cmd+Shift+Z for redo (alternative)
        if ((event.ctrlKey || event.metaKey) && event.shiftKey && event.key === 'z') {
          event.preventDefault()
          if (canRedo()) {
            onRedo()
          }
          return
        }

        // Delete or Backspace for delete (optional, if onDelete provided)
        if (onDelete && (event.key === 'Delete' || event.key === 'Backspace')) {
          event.preventDefault()
          onDelete()
          return
        }
      }

      onMounted(() => {
        window.addEventListener('keydown', handleKeydown)
      })

      onUnmounted(() => {
        window.removeEventListener('keydown', handleKeydown)
      })

      return {
        handleKeydown,
      }
    }
    ```

    Key points:
    - Handles Ctrl+Z/Cmd+Z for undo
    - Handles Ctrl+Y/Cmd+Y for redo
    - Handles Ctrl+Shift+Z for redo (alternative)
    - Checks if typing in input before triggering
    - Optional Delete key handler for selected elements
  </action>
