    1. Add localStorage persistence for styles:
    ```typescript
    // Add to existing script setup
    const STORAGE_KEY_STYLE = 'whiteboard:style'

    // Load saved styles or use defaults
    const savedStyles = localStorage.getItem(STORAGE_KEY_STYLE)
    const currentColor = ref(savedStyles ? JSON.parse(savedStyles).color : '#000000')
    const currentSize = ref(savedStyles ? JSON.parse(savedStyles).size : 4)

    // Save styles when changed
    watch([currentColor, currentSize], () => {
      localStorage.setItem(STORAGE_KEY_STYLE, JSON.stringify({
        color: currentColor.value,
        size: currentSize.value,
      }))
    })

    // Style change handlers
    function handleColorChange(color: string) {
      currentColor.value = color
    }

    function handleSizeChange(size: number) {
      currentSize.value = size
    }
    ```

    ```vue

      :currentColor="currentColor"

      @size-change="handleSizeChange"
    />
    ```
