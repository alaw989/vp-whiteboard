    ```typescript
    import { readMultipartFormData } from 'h3'
    import { nanoid } from 'nanoid'
    type { UploadResult } from '~/types'

    // Storage configuration - uses Supabase storage or local files
    // For MVP, using local file storage in public/uploads directory
    import fs from 'fs/promises'
    import path from 'path'

    const UPLOAD_DIR = path.join(process.cwd(), 'public', 'uploads')

    // Ensure upload directory exists
    async function ensureUploadDir() {
      try {
        await fs.access(UPLOAD_DIR)
      } catch {
        await fs.mkdir(UPLOAD_DIR, { recursive: true })
      }
    }

    export default defineEventHandler(async (event) => {
      try {
        await ensureUploadDir()

        // Parse multipart form data
        const formData = await readMultipartFormData(event)
        const file = formData.find(field => field.name === 'file')
        const whiteboardId = formData.find(field => field.name === 'whiteboard_id')?.data

        if (!file || !file.data || !file.filename) {
          return {
            success: false,
            error: 'No file uploaded',
          }
        }

        if (!whiteboardId) {
          return {
            success: false,
            error: 'No whiteboard ID provided',
          }
        }

        // Validate file type
        const allowedTypes = ['application/pdf', 'image/jpeg', 'image/png', 'image/webp']
        const fileType = file.type || 'application/octet-stream'

        if (!allowedTypes.includes(fileType)) {
          return {
            success: false,
            error: `Invalid file type. Allowed: ${allowedTypes.join(', ')}`,
          }
        }

        // Validate file size (10MB limit)
        const MAX_SIZE = 10 * 1024 * 1024
        if (file.data.length > MAX_SIZE) {
          return {
            success: false,
            error: `File size exceeds 10MB limit`,
          }
        }

        // Generate unique file ID and path
        const fileId = nanoid()
        const fileExtension = path.extname(file.filename)
        const safeFileName = `${fileId}${fileExtension}`
        const storagePath = path.join(UPLOAD_DIR, safeFileName)

        // Write file to disk
        await fs.writeFile(storagePath, Buffer.from(file.data))

        // Create file record (in production, save to database)
        const fileRecord = {
          id: fileId,
          whiteboard_id: new TextDecoder().decode(whiteboardId),
          file_name: file.filename,
          file_type: fileType,
          storage_path: `/uploads/${safeFileName}`,
          file_size: file.data.length,
          created_at: new Date().toISOString(),
        }

        // Return upload result
        const result: UploadResult = {
          fileId,
          fileName: file.filename,
          storagePath: `/uploads/${safeFileName}`,
          url: `/uploads/${safeFileName}`,
          fileRecord,
        }

        return {
          success: true,
          data: result,
        }
      } catch (error) {
        console.error('Upload error:', error)
        return {
          success: false,
          error: error instanceof Error ? error.message : 'Upload failed',
        }
      }
    })
    ```

    Key points:
    - Uses H3's readMultipartFormData for file parsing
    - Validates file type and size before processing
    - Generates unique file IDs using nanoid
    - Returns UploadResult compatible with frontend expectations
    - Stores files in public/uploads for MVP (Supabase storage for production)

    // Handle successful file upload - render document on canvas
    async function handleUploadSuccess(result: UploadResult) {
      if (!canvasRef.value) return

      try {
        const fileType = result.fileRecord?.file_type || ''

        if (fileType === 'application/pdf') {
          // For PDFs, fetch as ArrayBuffer and use addPDFLayer
          const response = await fetch(result.url)
          const arrayBuffer = await response.arrayBuffer()

          await canvasRef.value.addPDFLayer({
            id: result.fileId,
            url: result.url,
            name: result.fileName,
          }, arrayBuffer)
        } else {
          // For images, use addImageLayer directly
          await canvasRef.value.addImageLayer({
            id: result.fileId,
            url: result.url,
            name: result.fileName,
          })
        }
      } catch (error) {
        console.error('Failed to render document:', error)
        // Optionally show error to user
      }
    }

    /**
     * Bring layer to front (highest z-index)
     */
    function bringLayerToFront(layerId: string) {
      const layers = layerState.value.layers
      const index = layers.findIndex(l => l.id === layerId)
      if (index > -1) {
        const [layer] = layers.splice(index, 1)
        layers.push(layer)
      }
    }

    /**
     * Send layer to back (lowest z-index)
     */
    function sendLayerToBack(layerId: string) {
      const layers = layerState.value.layers
      const index = layers.findIndex(l => l.id === layerId)
      if (index > -1) {
        const [layer] = layers.splice(index, 1)
        layers.unshift(layer)
      }
    }

    /**
     * Toggle layer visibility
     */
    function toggleLayerVisibility(layerId: string) {
      const layer = layerState.value.layers.find(l => l.id === layerId)
      if (layer) {
        layer.visible = !layer.visible
      }
    }

    /**
     * Delete layer
     */
    function deleteLayer(layerId: string) {
      removeLayer(layerId)
    }
